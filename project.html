Swapping of Two Numbers
Run

Enter a value : 80

Enter b value : 40

Before swapping a = 80 ,b = 40

After swapping a = 40 ,b = 80

#include <stdio.h>
  int main() {
    int a,b,temp;
    printf("Enter a value : ");
    scanf("%d",&a);
    printf("Enter b value : ");
    scanf("%d",&b);
    printf("Before swapping a = %d ,b = %d \n",a,b);
    temp=a;
    a=b;
    b=temp;
    printf("After swapping a = %d ,b = %d",a,b);
    return 0;
    }
